  <h1>QR Code Generator</h1>

  <p>Enter the report number to generate

    placeholder="AGA-2026-00001">


  <button onclick="generateQR()">

    Generate QR Code

  <p class="print-info">

    Right-click the QR code → Save Image,

    then place it on your printed certificate.

  <!-- QR code library (free, open-source) -->

  <script src=

    "https://cdn.jsdelivr.net/npm/qrcodejs@1.0.0/qrcode.min.js">

  </script>

  <script>

    // IMPORTANT: Replace with YOUR actual site URL

    const SITE_URL =

      'https://yourusername.github.io/asian-gemological-association';


    function generateQR() {

      const reportNo =

        document.getElementById('reportInput')

          .value.trim();

      if (!reportNo) return;


      const container =

        document.getElementById('qrOutput');

      container.innerHTML = '';


      const verifyURL =

        SITE_URL + '/verify.html?id=' +

        encodeURIComponent(reportNo);


      new QRCode(container, {

        text: verifyURL,

        width: 200,

        height: 200,

        colorDark: '#1A1A2E',

        colorLight: '#FFFFFF'

      });


      container.innerHTML +=

        '<p style="margin-top:1rem;">' +

        '<strong>' + reportNo + '</strong>' +

        '<br><small>' + verifyURL +

        '</small></p>';

    }

  </script>
